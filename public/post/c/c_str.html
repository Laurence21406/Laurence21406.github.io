strspn strcspn strpbrk


/* strspn返回s起始部分匹配accept中任意字符的字符数，一旦从起始部分开始，不能从accept中找到匹配的字符，就截断，返回前面匹配的总的字符 */
size_t strspn(const char *s, const char *accept);
/* strspn返回s起始部分不匹配reject中任意字符的字符数 */
size_t strcspn(const char *s, const char *reject);

/* strpbrk在字符串s中找到与accept任何字符匹配的第一个字符（不包括终止空字符）*/
char *strpbrk(const char *s, const char *accept);

strstr的三种用法


/* strstr函数在第一个字符串haystack中搜索第二个字符串needle, 找到返回第二字符串needle首地址，未找到返回NULL*/
char *strstr(const char *haystack, const char *needle);
1.单次查找，返回查找成功的字符串以及后面的字符串 

void test1() {
  char *haystck = "The  strstr() function finds the first occurrence of the substring nee‐";
  char *needle  = "substring";
  char *p = strstr(haystck, needle);
  puts(p); // substring nee‐
}
2. 单次查找,返回子串在母串中位置 

void test2() {
  char *haystck = "The substring nee‐";
  char *needle  = "substring";
  char *p = strstr(haystck, needle);
  if(p != NULL) {
      ptrdiff_t pdf = p  - haystck;
  printf("%td\n", pdf);
 }
}
3. 多次查找,记录子串在母串中出现次数 

void test3() {
  char *haystck = "The substring nee‐substring， substring";
  char *needle  = "substring";
  int k = 0;
  char *p = haystck;
  size_t len = strlen(needle);
  while ((p = strstr(p, needle)) != NULL) {
    ++k;
    p += len;
  }
  printf("%d\n", k);
}
