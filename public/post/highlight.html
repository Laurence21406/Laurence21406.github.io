int main(int argc, char *argv[])
{
  pid_t pid;

  fflush(NULL);
  pid = fork();
  if (pid < 0) {
    perror("fork():");
    exit(EXIT_FAILURE);
  }

  if (pid == 0) {
    printf("[%d]:Child is working!\n",getpid());
  }else{
    printf("[%d]:Parent is working!\n",getpid());
  }
  printf("[%d]:End!\n",getpid());

  return 0;
}
