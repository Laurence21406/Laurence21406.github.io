SELECT department_id, MAX(salary)
FROM employee
WHERE department_id IN (10, 20, 30, 40) -- 过滤条件中没有聚合函数时，建议声明在WHERE子句中
GROUP BY department_id
HAVING MAX(salary) > 1000 -- 过滤条件中有聚合函数时, 过滤条件必须声明在HAVING子句中
